def create_workload(billing_group_id, name, zone, options={})
  unless options[:lease_expire_time].nil?
    options[:lease_expire_time] = options[:lease_expire_time].iso8601
  end

  request(
    :expects => [201],
    :method  => "POST",
    :path    => "v2/api/config/workload",
    :body    => Fog::JSON.encode({
      :billing_group => billing_group_id,
      :name          => name,
      :zone          => zone
    }.merge(options))
  )
end

def request(params)
  params[:method] ||= "GET"

  headers = params[:headers] || {}
  headers["Accept"]        = "application/json"
  headers["Content-type"]  = "application/json"
  headers["Authorization"] = build_auth(URI.join(api_host, params[:path]), params[:method])

  response = @connection.request(params.merge(
    :headers => headers
  ))

  unless response.body.empty?
    response.body = Fog::JSON.decode(response.body)
  end

  response
end

def update_server_template(id, attributes)
